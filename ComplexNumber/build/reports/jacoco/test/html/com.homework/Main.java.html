package com.homework;

public class Main {
    public static void main( String[] args ) {
        ComplexNumber cNumber1 = new ComplexNumber.Builder().real(2).im(3).build();
        ComplexNumber cNumber2 = new ComplexNumber.Builder().real(4).im(5).build();

        System.out.println("Summ of complex numbers "+cNumber1 +" and "+cNumber2 +" = "+ ComplexNumber.summ(cNumber1,cNumber2));
        System.out.println("Subtraction of complex numbers "+cNumber1 +" and "+cNumber2 +" = "+ ComplexNumber.sub(cNumber1,cNumber2));
        System.out.println("Multiplication of complex numbers "+cNumber1 +" and "+cNumber2 +" = "+ ComplexNumber.mult(cNumber1,cNumber2));
        System.out.println("The result of dividing a complex number "+cNumber1 +" by a complex number "+cNumber2 +" = "+ ComplexNumber.div(cNumber1,cNumber2));

    }
}
